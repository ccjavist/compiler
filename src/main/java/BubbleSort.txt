class BubbleSort is
    var arr: Array
    var n: Integer

    method sort(arr: Array) is
        this.arr := arr
        this.n := arr.length

        var i: Integer
        var j: Integer
        for i := 0 to n-1 loop
            for j := 0 to n-i-2 loop
                if arr[j] > arr[j+1] then
                    swap(j, j+1)
                end
            end
        end
    end

    method swap(i: Integer, j: Integer) is
        var temp: Integer
        temp := arr[i]
        arr[i] := arr[j]
        arr[j] := temp
    end
end
